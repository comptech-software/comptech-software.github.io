Classes & Objects

# Create a class
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def age_after_years(self, years):
        return "In {} years, I will be {} years old.".format(years, self.age + years)

# Create an object using our class
joe = Person("Joe", 25)

print(joe.name)
print(joe.age_after_years(5))
